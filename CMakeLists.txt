cmake_minimum_required(VERSION 3.16)

# Enable Objective-C++ for macOS Sparkle integration
if(APPLE)
  project(Phraims VERSION 0.55 LANGUAGES CXX OBJCXX)
else()
  project(Phraims VERSION 0.55 LANGUAGES CXX)
endif()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 6.9 COMPONENTS Widgets WebEngineWidgets Network REQUIRED)

# Generate version.h from version.h.in with current project version
configure_file(
  "${CMAKE_CURRENT_SOURCE_DIR}/version.h.in"
  "${CMAKE_CURRENT_BINARY_DIR}/version.h"
  @ONLY
)

# On macOS build a GUI app bundle (prevents launching a Terminal).
if(APPLE)
  #
  # From https://doc.qt.io/qt-6/appicon.html#using-cmake-on-macos
  #
  # The MACOSX_BUNDLE_ICON_FILE variable is added to the Info.plist
  # generated by CMake. This variable contains the .icns file name,
  # without the path.
  set(MACOSX_BUNDLE_ICON_FILE phraims.icns)

  # And the following tells CMake where to find and install the file itself.
  set(app_icon_macos "${CMAKE_CURRENT_SOURCE_DIR}/resources/phraims.icns")
  set_source_files_properties(${app_icon_macos} PROPERTIES MACOSX_PACKAGE_LOCATION "Resources")

  set(MACOSX_BUNDLE_EXECUTABLE_NAME Phraims)
  set(MACOSX_BUNDLE_ICON_FILE phraims.icns)
  set(MACOSX_BUNDLE_BUNDLE_NAME "${PROJECT_NAME}")
  set(MACOSX_BUNDLE_SHORT_VERSION_STRING "${PROJECT_VERSION}")
  set(MACOSX_BUNDLE_BUNDLE_VERSION "${PROJECT_VERSION}")
  set(CMAKE_OSX_DEPLOYMENT_TARGET "${CMAKE_OSX_DEPLOYMENT_TARGET}")

  configure_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/resources/Info.plist.in"
    "${CMAKE_CURRENT_BINARY_DIR}/Info.plist"
    @ONLY
  )

  qt_add_executable(Phraims
    MACOSX_BUNDLE
    main.cpp
    AppSettings.h
    Utils.h
    Utils.cpp
    EscapeFilter.h
    MyWebEnginePage.h
    MyWebEngineView.h
    SplitterDoubleClickFilter.h
    DomPatch.h
    DomPatch.cpp
    SplitFrameWidget.h
    SplitFrameWidget.cpp
    SplitWindow.h
    SplitWindow.cpp
    UpdateChecker.h
    UpdateChecker.cpp
    UpdateDialog.h
    UpdateDialog.cpp
    SparkleUpdater.h
    SparkleUpdater.mm
    resources/resources.qrc
    ${app_icon_macos}
  )
  # Mark SparkleUpdater.mm as Objective-C++ source
  set_source_files_properties(SparkleUpdater.mm PROPERTIES
    COMPILE_FLAGS "-x objective-c++"
  )
  
  set_target_properties(Phraims PROPERTIES
    MACOSX_BUNDLE TRUE
    MACOSX_BUNDLE_INFO_PLIST "${CMAKE_CURRENT_BINARY_DIR}/Info.plist"
    MACOSX_BUNDLE_SHORT_VERSION_STRING "${PROJECT_VERSION}"
    MACOSX_BUNDLE_BUNDLE_VERSION "${PROJECT_VERSION}"
    MACOSX_BUNDLE_BUNDLE_NAME "${PROJECT_NAME}"
  )

elseif(WIN32)
  # On Windows prefer a GUI subsystem build to avoid a console window.
  # This sets the WIN32_EXECUTABLE property so the linker uses the
  # WINDOWS subsystem instead of CONSOLE.
  
  # Configure the Windows resource file with version information
  # Extract version components for the .rc file
  # Note: This expects semantic versioning (MAJOR.MINOR[.PATCH]) without pre-release
  # or build metadata (e.g., "1.2.3" not "1.2.3-alpha" or "1.2.3+build")
  string(REGEX MATCH "^([0-9]+)\\.([0-9]+)(\\.([0-9]+))?" _ "${PROJECT_VERSION}")
  set(PHRAIMS_VERSION_MAJOR "${CMAKE_MATCH_1}")
  set(PHRAIMS_VERSION_MINOR "${CMAKE_MATCH_2}")
  set(PHRAIMS_VERSION_PATCH "${CMAKE_MATCH_4}")
  if(NOT PHRAIMS_VERSION_PATCH)
    set(PHRAIMS_VERSION_PATCH "0")
  endif()
  
  # Set the icon path for the resource script (absolute path required by RC.exe)
  set(PHRAIMS_ICON_PATH "${CMAKE_CURRENT_SOURCE_DIR}/resources/phraims.ico")
  
  configure_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/resources/phraims.rc.in"
    "${CMAKE_CURRENT_BINARY_DIR}/phraims.rc"
    @ONLY
  )
  
  qt_add_executable(Phraims
    main.cpp
    AppSettings.h
    Utils.h
    Utils.cpp
    EscapeFilter.h
    MyWebEnginePage.h
    MyWebEngineView.h
    SplitterDoubleClickFilter.h
    DomPatch.h
    DomPatch.cpp
    SplitFrameWidget.h
    SplitFrameWidget.cpp
    SplitWindow.h
    SplitWindow.cpp
    UpdateChecker.h
    UpdateChecker.cpp
    UpdateDialog.h
    UpdateDialog.cpp
    WinSparkleUpdater.h
    WinSparkleUpdater.cpp
    resources/resources.qrc
    "${CMAKE_CURRENT_BINARY_DIR}/phraims.rc"
  )

  # Ensure the executable is built as a GUI subsystem without a console window
  set_target_properties(Phraims PROPERTIES WIN32_EXECUTABLE TRUE)

else()
  # Linux and other platforms: standard executable build
  qt_add_executable(Phraims
    main.cpp
    AppSettings.h
    Utils.h
    Utils.cpp
    EscapeFilter.h
    MyWebEnginePage.h
    MyWebEngineView.h
    SplitterDoubleClickFilter.h
    DomPatch.h
    DomPatch.cpp
    SplitFrameWidget.h
    SplitFrameWidget.cpp
    SplitWindow.h
    SplitWindow.cpp
    UpdateChecker.h
    UpdateChecker.cpp
    UpdateDialog.h
    UpdateDialog.cpp
    resources/resources.qrc
  )

endif()

target_link_libraries(Phraims
  PRIVATE
    Qt6::Widgets
    Qt6::WebEngineWidgets
    Qt6::Network
)

# macOS: Link Sparkle framework if available
if(APPLE)

  # Try to find Sparkle framework in standard macOS locations
  find_library(SPARKLE_FRAMEWORK Sparkle
    PATHS
      ${CMAKE_BINARY_DIR}/Frameworks
  )
  
  if(SPARKLE_FRAMEWORK)
    message(STATUS "Found Sparkle framework: ${SPARKLE_FRAMEWORK}")
    target_link_libraries(Phraims PRIVATE ${SPARKLE_FRAMEWORK})
    
    # Copy Sparkle.framework into the app bundle
    # Use `ditto` to preserve symlinks
    add_custom_command(TARGET Phraims POST_BUILD
      COMMAND /bin/mkdir -p
        "$<TARGET_BUNDLE_DIR:Phraims>/Contents/Frameworks"
      COMMAND /usr/bin/ditto
        "${SPARKLE_FRAMEWORK}"
        "$<TARGET_BUNDLE_DIR:Phraims>/Contents/Frameworks/Sparkle.framework"
      COMMENT "Copying Sparkle.framework into app bundle with symlinks"
      VERBATIM
    )
  else()
    message(WARNING "Sparkle framework not found - macOS auto-updates will use manual download fallback")
  endif()

endif()

# Windows: Link WinSparkle library if available
if(WIN32)
  # Try to find WinSparkle library
  find_library(WINSPARKLE_LIBRARY WinSparkle
    PATHS
      ${CMAKE_SOURCE_DIR}/WinSparkle
      ${CMAKE_SOURCE_DIR}/lib
      "C:/Program Files/WinSparkle"
      "C:/Program Files (x86)/WinSparkle"
  )
  
  find_path(WINSPARKLE_INCLUDE_DIR winsparkle.h
    PATHS
      ${CMAKE_SOURCE_DIR}/WinSparkle/include
      ${CMAKE_SOURCE_DIR}/include
      "C:/Program Files/WinSparkle/include"
      "C:/Program Files (x86)/WinSparkle/include"
  )
  
  if(WINSPARKLE_LIBRARY AND WINSPARKLE_INCLUDE_DIR)
    message(STATUS "Found WinSparkle library: ${WINSPARKLE_LIBRARY}")
    message(STATUS "Found WinSparkle include dir: ${WINSPARKLE_INCLUDE_DIR}")
    target_include_directories(Phraims PRIVATE ${WINSPARKLE_INCLUDE_DIR})
    target_link_libraries(Phraims PRIVATE ${WINSPARKLE_LIBRARY})
    
    # Copy WinSparkle.dll next to the executable
    get_filename_component(WINSPARKLE_DIR ${WINSPARKLE_LIBRARY} DIRECTORY)
    find_file(WINSPARKLE_DLL WinSparkle.dll PATHS ${WINSPARKLE_DIR} ${WINSPARKLE_DIR}/../bin NO_DEFAULT_PATH)
    if(WINSPARKLE_DLL)
      add_custom_command(TARGET Phraims POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
          ${WINSPARKLE_DLL}
          $<TARGET_FILE_DIR:Phraims>
        COMMENT "Copying WinSparkle.dll to output directory"
      )
    endif()
  else()
    message(WARNING "WinSparkle library not found - Windows auto-updates will use manual download fallback")
  endif()
endif()

# Add build directory to include path so generated version.h can be found
target_include_directories(Phraims PRIVATE "${CMAKE_CURRENT_BINARY_DIR}")

if(APPLE)
  set(_qt_sdk "${CMAKE_OSX_SYSROOT}")
  if(_qt_sdk AND EXISTS "${_qt_sdk}/usr/include/c++/v1")
    target_include_directories(Phraims
      SYSTEM BEFORE
      PRIVATE
        "${_qt_sdk}/usr/include/c++/v1"
    )
  endif()
endif()
